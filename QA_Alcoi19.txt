def get_dose_rate_for_beam(beam):
    """Determina la tasa de dosis según la calidad del haz (FFF=1400, convencional=600)"""
    quality = beam.BeamQualityId
    return DOSE_RATE_MAP['FFF'] if "FFF" in quality.upper() else DOSE_RATE_MAP['default']

def derive_study_uid_from_plan_uid(plan_uid):
    """Deriva el StudyUID a partir del PlanUID (eliminando el último componente)"""
    if not plan_uid:
        return None
    parts = plan_uid.split('.')
    if len(parts) > 1:
        return '.'.join(parts[:-1])
    return plan_uid

def get_correct_study_uid():
    """Obtiene el StudyUID correcto derivado del plan de tratamiento original"""
    
    # Opción 1: Del beam_set actual (si es el de tratamiento, no QA)
    try:
        if hasattr(CURRENT_BEAMSET, 'ModificationInfo') and hasattr(CURRENT_BEAMSET.ModificationInfo, 'DicomUID'):
            beamset_name = CURRENT_BEAMSET.DicomPlanLabel
            if "QA" not in beamset_name and "EPID" not in beamset_name:
                plan_uid = CURRENT_BEAMSET.ModificationInfo.DicomUID
                study_uid = derive_study_uid_from_plan_uid(plan_uid)
                if study_uid:
                    return study_uid
    except Exception:
        pass
    
    # Opción 2: Del plan actual (el de tratamiento, NO el QA)
    try:
        if hasattr(CURRENT_PLAN, 'ModificationInfo') and hasattr(CURRENT_PLAN.ModificationInfo, 'DicomUID'):
            plan_name = CURRENT_PLAN.Name
            if "QA" not in plan_name and "EPID" not in plan_name:
                plan_uid = CURRENT_PLAN.ModificationInfo.DicomUID
                study_uid = derive_study_uid_from_plan_uid(plan_uid)
                if study_uid:
                    return study_uid
    except Exception:
        pass
    
    # Opción 3: Buscar en otros beam sets del plan que no sean QA
    try:
        for bs in CURRENT_PLAN.BeamSets:
            if hasattr(bs, 'ModificationInfo') and hasattr(bs.ModificationInfo, 'DicomUID'):
                if "QA" not in bs.DicomPlanLabel and "EPID" not in bs.DicomPlanLabel:
                    plan_uid = bs.ModificationInfo.DicomUID
                    study_uid = derive_study_uid_from_plan_uid(plan_uid)
                    if study_uid:
                        return study_uid
    except Exception:
        pass
    
    # Fallback: usar el UID del beam_set actual como base
    try:
        if hasattr(CURRENT_BEAMSET, 'ModificationInfo') and hasattr(CURRENT_BEAMSET.ModificationInfo, 'DicomUID'):
            plan_uid = CURRENT_BEAMSET.ModificationInfo.DicomUID
            study_uid = derive_study_uid_from_plan_uid(plan_uid)
            if study_uid:
                return study_uid
    except:
        pass

def create_octavius_qa():
    set_progress('Creando QA Octavius 1500...')
    print("\n🔬 Creando plan QA Octavius 1500...")
    
    existing_names = []
    if CURRENT_PLAN.VerificationPlans:
        for vp in CURRENT_PLAN.VerificationPlans:
            try:
                if hasattr(vp, 'BeamSet') and hasattr(vp.BeamSet, 'DicomPlanLabel'):
                    existing_names.append(vp.BeamSet.DicomPlanLabel)
            except:
                pass
    
    max_num = 0
    for name in existing_names:
        match = re.match(r'QA-Plan(\d+)', name)
        if match:
            num = int(match.group(1))
            if num > max_num:
                max_num = num
    candidate_num = max_num + 1
    qa_name = f"QA-Plan{candidate_num}"
    while qa_name in existing_names:
        candidate_num += 1
        qa_name = f"QA-Plan{candidate_num}"
    
    dose_grid = CURRENT_BEAMSET.GetDoseGrid()
    voxel_size = dose_grid.VoxelSize.x
    print(f"  Nombre: {qa_name}, Isocentro: {OCTAVIUS_ISOCENTER}, Voxel: {voxel_size} mm")
    
    try:
        CURRENT_BEAMSET.CreateQAPlan(
            PhantomName=OCTAVIUS_PHANTOM_NAME,
            PhantomId=OCTAVIUS_PHANTOM_ID,
            QAPlanName=qa_name,
            IsoCenter=OCTAVIUS_ISOCENTER,
            DoseGrid={'x': voxel_size, 'y': voxel_size, 'z': voxel_size},
            GantryAngle=None,
            CollimatorAngle=None,
            CouchRotationAngle=0,
            ComputeDoseWhenPlanIsCreated=True,
            RemoveCompensators=False,
            EnableDynamicTracking=False
        )
        save_all()
        set_progress(f'✅ QA Octavius creado: {qa_name}')
        print(f"  ✅ QA creado: {qa_name}")
        return qa_name, candidate_num
    except Exception as e:
        set_progress(f'❌ Error: {e}')
        print(f"  ❌ Error: {e}")
        return None, None

def get_interpolated_dose(detector_plane_y, dosegrid, doses):
    global DIMENS
    DIMENS = doses[0].DoseValues.DoseData.shape
    
    voxel_center_xmin = dosegrid.Corner.x + dosegrid.VoxelSize.x / 2
    voxel_center_ymin = dosegrid.Corner.y + dosegrid.VoxelSize.y / 2
    voxel_center_zmin = dosegrid.Corner.z + dosegrid.VoxelSize.z / 2
    voxel_center_xmax = voxel_center_xmin + (dosegrid.NrVoxels.x - 1) * dosegrid.VoxelSize.x
    voxel_center_ymax = voxel_center_ymin + (dosegrid.NrVoxels.y - 1) * dosegrid.VoxelSize.y
    voxel_center_zmax = voxel_center_zmin + (dosegrid.NrVoxels.z - 1) * dosegrid.VoxelSize.z
    
    xlin = np.linspace(voxel_center_xmin, voxel_center_xmax, dosegrid.NrVoxels.x)
    ylin = np.linspace(voxel_center_ymin, voxel_center_ymax, dosegrid.NrVoxels.y)
    zlin = np.linspace(voxel_center_zmin, voxel_center_zmax, dosegrid.NrVoxels.z)
    
    points = []
    for z in np.flipud(zlin):
        for x in xlin:
            points.append([z, detector_plane_y, x])
    points = np.array(points)
    
    dose_planes = []
    for dose in doses:
        interp_func = RegularGridInterpolator((zlin, ylin, xlin), dose.DoseValues.DoseData)
        dose_planes.append(interp_func(points))
    return np.array(dose_planes, dtype=object)

def export_epid_dicom(export_path, dicom_files, planar_doses):
    set_progress('Exportando imágenes EPID (cGy, factor 0.0046)...')
    print("\n📸 Exportando imágenes EPID...")
    global DIMENS
    if DIMENS is None:
        set_progress('❌ Error: DIMENS no definido')
        raise Exception("DIMENS no está definido.")
    
    nz = DIMENS[0]
    nx = DIMENS[2]
    factor = 0.01046  # sin filtro
    
    for i, dose_file in enumerate(dicom_files):
        set_progress(f'Imagen {i+1}/{len(dicom_files)}: {dose_file.RTImageLabel}')
        dose_gy = planar_doses[i]
        dose_cgy = dose_gy * factor
        max_dose = np.max(dose_cgy)
        rescale_slope = max_dose / 65535.0 if max_dose != 0 else 1.0
        pixel_values = (dose_cgy / rescale_slope).astype(np.uint16)
        if np.max(pixel_values) > 65535:
            pixel_values = np.clip(pixel_values, 0, 65535)
        dose_file.PixelData = pixel_values.tobytes()
        dose_file.RescaleSlope = str(rescale_slope)
        dose_file.RescaleIntercept = "0.0"
        dose_file.RescaleType = ''
        dose_file.DoseUnits = ''
        max_pixel = np.max(pixel_values)
        min_pixel = np.min(pixel_values)
        dose_file.WindowCenter = str((max_pixel + min_pixel) / 2.0)
        dose_file.WindowWidth = str(max_pixel - min_pixel)
        
        center_idx = (nz // 2) * nx + (nx // 2)
        dose_centro = dose_cgy[center_idx]
        pixel_centro = pixel_values[center_idx]
        
        file_name = f'EPID_{dose_file.RTImageLabel}'
        for char in '<>:"/\\|?*':
            file_name = file_name.replace(char, '_')
        output_path = os.path.join(export_path, f'{file_name}.dcm')
        dose_file.save_as(output_path, write_like_original=False)
        print(f"  ✅ {os.path.basename(output_path)} (centro {dose_centro:.4f} cGy)")
    set_progress('✅ Imágenes EPID exportadas')

def get_verification_export_path(patient_id):
    patient_root = os.path.join(VERIFIC_ROOT, patient_id)
    
    if not os.path.exists(patient_root):
        os.makedirs(patient_root)
        return patient_root
    
    subfolders = []
    for name in os.listdir(patient_root):
        full_path = os.path.join(patient_root, name)
        if os.path.isdir(full_path):
            try:
                subfolders.append(int(name))
            except:
                pass
    
    next_num = max(subfolders) + 1 if subfolders else 1
    export_path = os.path.join(patient_root, str(next_num))
    os.makedirs(export_path)
    return export_path

def patch_epid_rtplan_with_correct_dose_rates(patient_info):
    """Parchea el RTPLAN del QA EPID con las tasas de dosis correctas (600/1400 según calidad)"""
    set_progress('Parcheando RTPLAN del QA EPID con tasas correctas...')
    print("\n📝 Parcheando RTPLAN del QA EPID...")
    
    # Buscar archivos RTPLAN en la carpeta ARIA
    plans = glob.glob(os.path.join(ARIA_PATH, "RP*.dcm"))
    
    if not plans or not patient_info.get('StudyInstanceUID'):
        print("  ⚠️ No se encontraron archivos RTPLAN")
        return False
    
    # Seleccionar el archivo RTPLAN más reciente (debería ser el QA EPID)
    latest_plan = max(plans, key=os.path.getctime)
    print(f"  📄 Modificando: {os.path.basename(latest_plan)}")
    
    # Crear diccionario de tasas según calidad del haz del plan original
    dose_rates = {}
    print("  📊 Tasas a asignar según calidad de haz:")
    for beam in CURRENT_BEAMSET.Beams:
        rate = get_dose_rate_for_beam(beam)
        dose_rates[beam.Number] = rate
        print(f"    ✓ Haz {beam.Number} ({beam.Name}, {beam.BeamQualityId}): {rate} MU/min")
    
    try:
        # Leer el archivo DICOM
        ds = pydicom.dcmread(latest_plan)
        modified = False
        
        # Actualizar StudyInstanceUID si es necesario
        if hasattr(ds, 'StudyInstanceUID') and ds.StudyInstanceUID != patient_info.get('StudyInstanceUID'):
            ds.StudyInstanceUID = patient_info.get('StudyInstanceUID')
            modified = True
            print(f"    ✓ StudyInstanceUID actualizado")
        
        # Actualizar PatientID si es necesario
        if hasattr(ds, 'PatientID') and ds.PatientID != patient_info['PatientID']:
            ds.PatientID = patient_info['PatientID']
            modified = True
            print(f"    ✓ PatientID actualizado")
        
        # Actualizar PatientName si es necesario
        if hasattr(ds, 'PatientName') and ds.PatientName != patient_info['PatientName']:
            ds.PatientName = patient_info['PatientName']
            modified = True
            print(f"    ✓ PatientName actualizado")
        
        # Actualizar etiqueta del plan
        if hasattr(ds, 'RTPlanLabel') and ds.RTPlanLabel != "QA PORTAL":
            ds.RTPlanLabel = "QA PORTAL"
            modified = True
            print(f"    ✏️ RTPlanLabel → QA PORTAL")
        
        # Recorrer los haces y asignar las tasas
        if hasattr(ds, 'BeamSequence'):
            for beam in ds.BeamSequence:
                beam_num = getattr(beam, 'BeamNumber', None)
                
                if beam_num in dose_rates:
                    tasa_correcta = dose_rates[beam_num]
                    
                    # Modificar DoseRate en el haz
                    if hasattr(beam, 'DoseRate'):
                        old_rate = beam.DoseRate
                        if str(old_rate) != str(tasa_correcta):
                            beam.DoseRate = tasa_correcta
                            modified = True
                            print(f"    ✏️ Haz {beam_num}: DoseRate {old_rate} → {tasa_correcta} MU/min")
                    
                    # Modificar DoseRateSet en los puntos de control
                    if hasattr(beam, 'ControlPointSequence'):
                        for cp_idx, cp in enumerate(beam.ControlPointSequence):
                            if hasattr(cp, 'DoseRateSet'):
                                old_rate = cp.DoseRateSet
                                if str(old_rate) != str(tasa_correcta):
                                    cp.DoseRateSet = tasa_correcta
                                    modified = True
                                    print(f"      ↻ Haz {beam_num} CP{cp_idx+1}: DoseRateSet {old_rate} → {tasa_correcta} MU/min")
                    
                    # Modificar SegmentDoseRate si existe (para IMRT)
                    if hasattr(beam, 'BeamSegmentSequence'):
                        for seg_idx, seg in enumerate(beam.BeamSegmentSequence):
                            if hasattr(seg, 'SegmentDoseRate'):
                                old_rate = seg.SegmentDoseRate
                                if str(old_rate) != str(tasa_correcta):
                                    seg.SegmentDoseRate = tasa_correcta
                                    modified = True
                                    print(f"      ↻ Haz {beam_num} Segmento {seg_idx+1}: SegmentDoseRate {old_rate} → {tasa_correcta} MU/min")
                else:
                    print(f"    ⚠️ Haz {beam_num}: no se encontró en el plan original")
        
        # Guardar los cambios
        if modified:
            ds.save_as(latest_plan)
            print(f"\n  ✅ RTPLAN actualizado con tasas correctas (600/1400)")
            set_progress('✅ Tasas de dosis corregidas en RTPLAN')
            return True
        else:
            print(f"\n  ✓ RTPLAN ya tenía las tasas correctas")
            return True
            
    except Exception as e:
        print(f"  ❌ Error: {e}")
        set_progress(f'❌ Error actualizando RTPLAN: {e}')
        return False

def patch_rtimages_with_correct_uid(export_path, patient_info):
    """Parchea las imágenes RTImage exportadas con los UIDs correctos"""
    set_progress('Parcheando RTImages...')
    
    rtimages = glob.glob(os.path.join(export_path, "EPID_*.dcm"))
    
    if not rtimages:
        return
    
    correct_study_uid = patient_info.get('StudyInstanceUID')
    correct_patient_id = patient_info['PatientID']
    correct_patient_name = patient_info['PatientName']
    correct_plan_uid = patient_info.get('OriginalPlanSOPInstanceUID')
    
    if not correct_study_uid:
        return
    
    for f in rtimages:
        try:
            ds = pydicom.dcmread(f)
            modified = False
            
            if hasattr(ds, 'StudyInstanceUID') and ds.StudyInstanceUID != correct_study_uid:
                ds.StudyInstanceUID = correct_study_uid
                modified = True
            
            if hasattr(ds, 'PatientID') and ds.PatientID != correct_patient_id:
                ds.PatientID = correct_patient_id
                modified = True
            
            if hasattr(ds, 'PatientName') and ds.PatientName != correct_patient_name:
                ds.PatientName = correct_patient_name
                modified = True
            
            if hasattr(ds, 'ReferencedRTPlanSequence') and correct_plan_uid:
                for ref in ds.ReferencedRTPlanSequence:
                    if hasattr(ref, 'ReferencedSOPInstanceUID'):
                        if ref.ReferencedSOPInstanceUID != correct_plan_uid:
                            ref.ReferencedSOPInstanceUID = correct_plan_uid
                            modified = True
            
            if modified:
                ds.save_as(f)
        except Exception:
            pass
    
    set_progress('✅ RTImages parcheadas')

def create_epid_qa_plan():
    set_progress('Creando plan EPID QA...')
    print("\n🔬 Creando plan QA EPID Varian...")
    
    existing_names = []
    if CURRENT_PLAN.VerificationPlans:
        for vp in CURRENT_PLAN.VerificationPlans:
            try:
                if hasattr(vp, 'BeamSet') and hasattr(vp.BeamSet, 'DicomPlanLabel'):
                    existing_names.append(vp.BeamSet.DicomPlanLabel)
            except:
                pass
    
    max_num = 0
    for name in existing_names:
        match = re.match(r'EPID QA (\d+)', name)
        if match:
            num = int(match.group(1))
            if num > max_num:
                max_num = num
    candidate_num = max_num + 1
    qa_name = f"EPID QA {candidate_num}"
    while qa_name in existing_names:
        candidate_num += 1
        qa_name = f"EPID QA {candidate_num}"
    
    print(f"  Nombre: {qa_name}")
    epid_options = {
        'PhantomName': PHANTOM_NAME,
        'PhantomId': PHANTOM_ID,
        'DoseGrid': GRID_RESOLUTION,
        'GantryAngle': 0,
        'CouchRotationAngle': 0,
        'ApplyFloodFieldCorrection': False,
        'SingleFloodFieldBeamQualityId': None,
        'FloodFieldFactor': 0.5,
        'IsoCenter': ISOCENTER,
        'QAPlanName': qa_name
    }
    if COLLIMATOR_ANGLE != '':
        epid_options['CollimatorAngle'] = COLLIMATOR_ANGLE
    
    try:
        CURRENT_BEAMSET.CreateEPIDResponse(**epid_options)
        set_progress(f'✅ EPID QA creado: {qa_name}')
        print(f"  ✅ Plan EPID QA creado: {qa_name}")
    except Exception as e:
        set_progress(f'❌ Error: {e}')
        raise Exception(f'Error creando plan EPID QA: {e}')

    dose_grid = epid_qa_plan.BeamSet.FractionDose.InDoseGrid
